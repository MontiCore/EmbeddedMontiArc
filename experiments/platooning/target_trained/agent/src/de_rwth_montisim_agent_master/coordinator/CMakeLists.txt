
cmake_minimum_required(VERSION 3.5)
project (Coordinator_de_rwth_montisim_agent_master LANGUAGES CXX)

set (CMAKE_CXX_STANDARD 14)
set (THREADS_PREFER_PTHREAD_FLAG ON)
find_package(Threads REQUIRED)
add_library(IAdapter_de_rwth_montisim_agent_master IAdapter_de_rwth_montisim_agent_master.cpp)
target_link_libraries(IAdapter_de_rwth_montisim_agent_master de_rwth_montisim_agent_master)
target_include_directories(IAdapter_de_rwth_montisim_agent_master PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})

add_executable(Coordinator_de_rwth_montisim_agent_master Coordinator_de_rwth_montisim_agent_master.cpp)
set_target_properties(Coordinator_de_rwth_montisim_agent_master PROPERTIES LINKER_LANGUAGE CXX)
target_link_libraries(Coordinator_de_rwth_montisim_agent_master RosAdapter_de_rwth_montisim_agent_master de_rwth_montisim_agent_master Threads::Threads IAdapter_de_rwth_montisim_agent_master)
target_include_directories(Coordinator_de_rwth_montisim_agent_master PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})

export(TARGETS Coordinator_de_rwth_montisim_agent_master IAdapter_de_rwth_montisim_agent_master FILE Coordinator_de_rwth_montisim_agent_master.cmake)

install(TARGETS Coordinator_de_rwth_montisim_agent_master DESTINATION bin)
